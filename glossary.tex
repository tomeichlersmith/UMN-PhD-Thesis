\newglossaryentry{dm-full}{
    name={Dark Matter},
    description={Hypothesized particles that interact gravitationally with standard matter.}
}
\newacronym{dm}{DM}{Dark Matter}

\newglossaryentry{gr-full}{
    name={Einstein's Theory of General Relativity},
    description={Einstein's description of gravity based on the warping of spacetime}
}
\newacronym{gr}{GR}{Einstein's Theory of General Relativity}

\newglossaryentry{grav-lens}{
    name={Gravitational Lensing},
    description={The bending of light's path due to the presence of massive objects.}
}

\newglossaryentry{cmb-full}{
    name={Cosmic Microwave Background},
    description={Relic light from the early universe that was initially freed when matter cooled out of a dense plasma.}
}
\newacronym{cmb}{CMB}{\gls{cmb-full}}

\newglossaryentry{bao-full}{
    name={Baryon Acoustic Oscilations},
    description={}
}

\newglossaryentry{dark-photon}{
    name={dark photon},
    description={Postulated massive gauge boson for an additional $U_D(1)$ symmetry of nature.
            Often denoted as \(A'\).}
}

\newglossaryentry{hep-full}{
    name={High Energy Physics},
    description={The study of physics at high energies which, at these scales, is described by particles and their interactions.}
}
\newacronym{hep}{HEP}{\gls{hep-full}}

\newglossaryentry{ldmx-full}{
    name={Light Dark Matter eXperiment},
    description={A proposed fixed target experiment aiming to search critical light dark matter phase space.}
}
\newacronym{ldmx}{LDMX}{\gls{ldmx-full}}

\newglossaryentry{hps-full}{
    name={Heavy Photon Search experiment},
